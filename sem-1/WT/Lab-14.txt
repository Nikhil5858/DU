2. Script to add a click event handler to list items, change background color, slide up, and fade out

Copy code
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>jQuery List Item Click</title>
    <script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>
    <style>
        ul {
            list-style-type: none;
        }

        li {
            padding: 10px;
            border: 1px solid black;
            margin: 5px 0;
        }
    </style>
</head>
<body>

    <ul id="myList">
        <li>Item 1</li>
        <li>Item 2</li>
        <li>Item 3</li>
    </ul>

    <script>
        $(document).ready(function() {
            $('#myList li').click(function() {
                $(this).css('background-color', 'yellow').slideUp(500).fadeOut(500);
            });
        });
    </script>

</body>
</html>
