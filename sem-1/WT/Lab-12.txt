2. Recognize which mouse event is fired:
html
Copy code
<!DOCTYPE html>
<html>
<head>
    <title>Mouse Event</title>
</head>
<body>
    <div id="mouseEventDiv" style="height:200px; background-color:lightgrey;">Mouse over me!</div>
    <p id="mouseEventInfo"></p>

    <script>
        const mouseEventDiv = document.getElementById("mouseEventDiv");
        const mouseEventInfo = document.getElementById("mouseEventInfo");

        ["click", "dblclick", "mousedown", "mouseup", "mouseover", "mouseout"].forEach(eventType => {
            mouseEventDiv.addEventListener(eventType, function(event) {
                mouseEventInfo.innerText = `Mouse Event Fired: ${event.type}`;
            });
        });
    </script>
</body>
</html>

4. Recognize which form event is fired:
html
Copy code
<!DOCTYPE html>
<html>
<head>
    <title>Form Event</title>
</head>
<body>
    <form id="sampleForm">
        <input type="text" id="name" placeholder="Enter Name"><br><br>
        <input type="submit" value="Submit">
    </form>
    <p id="formEventInfo"></p>

    <script>
        const sampleForm = document.getElementById("sampleForm");
        const formEventInfo = document.getElementById("formEventInfo");

        ["submit", "focus", "blur", "input", "change"].forEach(eventType => {
            sampleForm.addEventListener(eventType, function(event) {
                formEventInfo.innerText = `Form Event Fired: ${event.type}`;
                event.preventDefault();
            }, true);
        });
    </script>
</body>

6. Student registration form with validation:
html
Copy code
<!DOCTYPE html>
<html>
<head>
    <title>Student Registration Form</title>
</head>
<body>
    <form id="registrationForm">
        <label>Name: <input type="text" id="name"></label><br><br>
        <label>Email: <input type="email" id="email"></label><br><br>
        <label>Age: <input type="number" id="age"></label><br><br>
        <input type="submit" value="Register">
    </form>
    <p id="errorMsg"></p>

    <script>
        const form = document.getElementById("registrationForm");
        form.addEventListener("submit", function(event) {
            const name = document.getElementById("name").value;
            const email = document.getElementById("email").value;
            const age = document.getElementById("age").value;
            const errorMsg = document.getElementById("errorMsg");

            if (name === "" || email === "" || age === "") {
                errorMsg.innerText = "All fields are required.";
                event.preventDefault();
            } else {
                errorMsg.innerText = "";
                alert("Form submitted successfully!");
            }
        });
    </script>
</body>
</html>



7. CRUD operation on Array using HTML and JavaScript:
html
Copy code
<!DOCTYPE html>
<html>
<head>
    <title>CRUD on Array</title>
</head>
<body>
    <input type="text" id="itemInput" placeholder="Enter item">
    <button onclick="addItem()">Add Item</button>
    <ul id="itemList"></ul>

    <script>
        let items = [];

        function addItem() {
            const item = document.getElementById("itemInput").value;
            items.push(item);
            displayItems();
        }

        function deleteItem(index) {
            items.splice(index, 1);
            displayItems();
        }

        function updateItem(index) {
            const newItem = prompt("Update item:", items[index]);
            if (newItem) {
                items[index] = newItem;
                displayItems();
            }
        }

        function displayItems() {
            const itemList = document.getElementById("itemList");
            itemList.innerHTML = "";
            items.forEach((item, index) => {
                itemList.innerHTML += `<li>${item} 
                    <button onclick="updateItem(${index})">Update</button> 
                    <button onclick="deleteItem(${index})">Delete</button></li>`;
            });
        }
    </script>
</body>
</html>



8. Handling mouse events:
html
Copy code
<!DOCTYPE html>
<html>
<head>
    <title>Mouse Events</title>
</head>
<body>
    <h1 id="heading">Mouse Over Me</h1>
    <button id="findTimeBtn">Find Time</button>
    <button id="redBtn">Red</button>
    <button id="greenBtn">Green</button>
    <p id="timeInfo"></p>

    <script>
        const heading = document.getElementById("heading");
        heading.onmouseover = () => heading.style.color = "yellow";
        heading.onmouseout = () => heading.style.color = "black";

        document.getElementById("findTimeBtn").onclick = () => {
            const date = new Date();
            document.getElementById("timeInfo").innerText = `Time: ${date.toLocaleTimeString()}, Date: ${date.toLocaleDateString()}`;
        };

        document.getElementById("redBtn").onclick = () => document.body.style.backgroundColor = "red";
        document.getElementById("greenBtn").onclick = () => document.body.style.backgroundColor = "green";
    </script>
</body>
</html>

10. Vowel detection and background change:
html
Copy code
<!DOCTYPE html>
<html>
<head>
    <title>Vowel Detection</title>
</head>
<body>
    <input type="text" id="vowelInput" placeholder="Press a key">
    <p id="vowelInfo"></p>

    <script>
        const vowels = ["a", "e", "i", "o", "u"];
        const vowelInput = document.getElementById("vowelInput");

        vowelInput.addEventListener("keydown", function(event) {
            const keyPressed = event.key.toLowerCase();
            if (vowels.includes(keyPressed)) {
                document.getElementById("vowelInfo").innerText = `Vowel Pressed: ${keyPressed}`;
            } else {
                document.getElementById("vowelInfo").innerText = "";
            }
        });

        vowelInput.addEventListener("keyup", function() {
            document.body.style.backgroundColor = "blue";
        });
    </script>
</body>
</html>
